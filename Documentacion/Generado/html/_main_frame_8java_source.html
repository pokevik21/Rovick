227  public void removeTime(String move){
228  int seg = 0;
229  if(Utiles.tieneNumero(move)){
230  String lastMoveSinNum = move.substring(0, Utiles.primerNum(move));
231  int numerMovs = Utiles.extraerNumero(move);
232  switch (lastMoveSinNum) {
233  case "F":
234  case "FD":
235  case "B":
236  case "BD":
237  seg = 6;
238  break;
239  }
240  this.numMovimientos -= numerMovs;
241  seg += 2*numerMovs;
242  }else{
243  switch (move) {
244  case "F":
245  case "FD":
246  case "B":
247  case "BD":
248  seg = 8;
249  break;
250  default:
251  seg = 2;
252  }
253  this.numMovimientos --;
254  }
255  tiempo.add(GregorianCalendar.SECOND, -1*seg);
256  }
