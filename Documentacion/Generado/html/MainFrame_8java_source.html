157  public void addTime(String tipoMove){

159  switch (tipoMove) {

161  case "FD":
162  case "B":
163  case "BD":

212  public void removeTime(String move){
213  int seg = 0;
214  if(Utiles.tieneNumero(move)){
215  String lastMoveSinNum = move.substring(0, Utiles.primerNum(move));
216  int numerMovs = Utiles.extraerNumero(move);
217  switch (lastMoveSinNum) {
218  case "F":
219  case "FD":
220  case "B":
221  case "BD":
222  seg = 6;
223  break;
224  }
225  this.numMovimientos -= numerMovs;
226  seg += 2*numerMovs;
227  }else{
228  switch (move) {
229  case "F":
230  case "FD":
231  case "B":
232  case "BD":
233  seg = 8;
234  break;
235  default:
236  seg = 2;
237  }
238  this.numMovimientos --;
239  }
240  tiempo.add(GregorianCalendar.SECOND, -1*seg);
241  }
