277  public void removeTime(String move){
278  int seg = 0;
279  if(Utiles.tieneNumero(move)){
280  String lastMoveSinNum = move.substring(0, Utiles.primerNum(move));
281  int numerMovs = Utiles.extraerNumero(move);
282  switch (lastMoveSinNum) {
283  case "F":
284  case "FD":
285  case "B":
286  case "BD":
287  seg = 6;
288  break;
289  }
290  this.numMovimientos -= numerMovs;
291  seg += 2*numerMovs;
292  }else{
293  switch (move) {
294  case "F":
295  case "FD":
296  case "B":
297  case "BD":
298  seg = 8;
299  break;
300  default:
301  seg = 2;
302  }
303  this.numMovimientos --;
304  }
305  tiempo.add(GregorianCalendar.SECOND, -1*seg);
306  }
