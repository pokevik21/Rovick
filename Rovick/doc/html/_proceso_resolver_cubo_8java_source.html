109  private float distancia(float[] c1, float[] c2) {
110  float dR = (float) Math.abs(c1[0]-c2[0]);
111  float dG = (float) Math.abs(c1[1]-c2[1]);
112  float dB = (float) Math.abs(c1[2]-c2[2]);
113  float resultado = (float) Math.sqrt(2*Math.pow(dR, 2) + 4*Math.pow(dG, 2) + 3*Math.pow(dB, 2));
